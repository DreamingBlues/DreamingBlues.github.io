Scans the picture horizontally, and looks for white space to seperate the characters

vertical_sums = np.sum(binary, axis=0)
char_start = []
char_end = []

in_character = False
for x, col_sum in enumerate(vertical_sums):
    if col_sum > 0 and not in_character:
        char_start.append(x)
        in_character = True
    elif col_sum == 0 and in_character:
        char_end.append(x)
        in_character = False

Extract and save characters¶

In [ ]:
os.makedirs("characters", exist_ok=True)

for i, (start, end) in enumerate(zip(char_start, char_end)):
    char_image = binary[:, start:end]
    if char_image.shape[1] > 0:
        # Trim whitespace
        coords = cv2.findNonZero(char_image)
        x, y, w, h = cv2.boundingRect(coords)
        char_crop = char_image[y:y+h, x:x+w]

        # Resize to fit in 45x45 with preserved aspect ratio, NO anti-aliasing
        h_new, w_new = char_crop.shape
        scale = 45.0 / max(h_new, w_new)
        new_w, new_h = int(w_new * scale), int(h_new * scale)
        char_resized = cv2.resize(char_crop, (new_w, new_h), interpolation=cv2.INTER_NEAREST)

        # Pad to center in 45x45 box
        pad_top = (45 - new_h) // 2
        pad_bottom = 45 - new_h - pad_top
        pad_left = (45 - new_w) // 2
        pad_right = 45 - new_w - pad_left
        char_padded = np.pad(char_resized, ((pad_top, pad_bottom), (pad_left, pad_right)), 'constant', constant_values=0)

        # Invert to match dataset style (white character on black background)
        char_padded = 255 - char_padded
        char_padded = char_padded.astype(np.uint8)

        save_path = f"characters/character_{i}.png"
        cv2.imwrite(save_path, char_padded)

        # Display each character
        plt.figure(figsize=(2, 2))
        plt.imshow(char_padded, cmap='gray')
        plt.title(f"Character {i}")
        plt.axis('off')
        plt.show()

data_dir = './dataset'  # folder with subfolders per class
img_size = (45, 45)
output_npz = 'processed_dataset.npz'


# Load from .npz if it exists, otherwise preprocess
if os.path.exists(output_npz):
    print(f"Loading preprocessed data from: {output_npz}")
    data = np.load(output_npz)
    X_train = data['X_train']
    X_val = data['X_val']
    y_train = data['y_train']
    y_val = data['y_val']
    class_names = data['class_names']
    print(f"Loaded shapes: X_train={X_train.shape}, X_val={X_val.shape}")
else:
    print("Preprocessing dataset...")

    images = []
    labels = []
    class_names = sorted([d for d in os.listdir(data_dir) if os.path.isdir(os.path.join(data_dir, d))])

    for class_idx, class_name in enumerate(class_names):
        class_path = os.path.join(data_dir, class_name)
        for fname in os.listdir(class_path):
            fpath = os.path.join(class_path, fname)
            try:
                img = load_img(fpath, target_size=img_size, color_mode='grayscale')
                img_array = img_to_array(img) / 255.0
                images.append(img_array)
                labels.append(class_idx)
            except Exception as e:
                print(f"Error loading {fpath}: {e}")

    X = np.array(images, dtype='float32')
    y = to_categorical(np.array(labels), num_classes=len(class_names))

    X_train, X_val, y_train, y_val = train_test_split(
        X, y, test_size=0.2, stratify=y, random_state=42
    )

    np.savez_compressed(output_npz,
                        X_train=X_train, X_val=X_val,
                        y_train=y_train, y_val=y_val,
                        class_names=np.array(class_names))
    print(f"Saved preprocessed dataset to: {output_npz}")

model = Sequential([

# Predict and decode characters using cnn_model.keras and 45x45 images
expression = ""
for fname in character_files:
    img_path = os.path.join(character_dir, fname)
    img = image.load_img(img_path, color_mode='grayscale', target_size=(45, 45))
    img_array = image.img_to_array(img) / 255.0
    if img_array.shape[-1] == 1:
        img_input = np.expand_dims(img_array, axis=0)
    else:
        img_input = np.expand_dims(img_array[..., 0], axis=(0, -1))

    prediction = model.predict(img_input)[0]
    pred_index = int(np.argmax(prediction))
    pred_label = index_to_label[pred_index]
    confidence = float(np.max(prediction)) * 100
    expression += pred_label

    # Top 3 predictions with confidence
    top_indices = prediction.argsort()[-3:][::-1]
    top_preds = [(index_to_label[i], prediction[i] * 100) for i in top_indices]

    # Display
    plt.figure(figsize=(2, 2)) 
    plt.imshow(img, cmap='gray')
    plt.title(f"Predicted: {pred_label} ({confidence:.2f}%)")
    plt.axis('off')
    plt.show()

    print(f"{fname}: {pred_label} ({confidence:.2f}%)")
    print("Top 3 Predictions:")
    for lbl, conf in top_preds:
        print(f" - {lbl}: {conf:.2f}%")
    print("--------------------")

print("Recognized Expression:", expression)

import requests
import xml.etree.ElementTree as ET

app_id = 'TW5YV3-G93X3JEQXK'
query = 'solve' + expression

# Make request to Wolfram Alpha API
url = 'https://api.wolframalpha.com/v2/query'
params = {
    'input': query,
    'appid': app_id,
    'format': 'plaintext'
}
response = requests.get(url, params=params)

# Parse the XML response
root = ET.fromstring(response.content)

# Print all pods and their content
found = False
for pod in root.findall('.//pod'):
    title = pod.attrib.get('title', '')
    plaintext = pod.find('.//plaintext')
    if plaintext is not None and plaintext.text:
        print(f"{title}:\n{plaintext.text}\n")
        found = True

if not found:
    print("No plaintext result found.")
